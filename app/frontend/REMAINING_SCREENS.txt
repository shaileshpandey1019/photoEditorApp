/**
 * Create Screen - Photo Collage Maker
 * Collage editor with photo selection and canvas
 */

import React, { useState, useRef } from 'react';
import {
  View,
  Text,
  StyleSheet,
  ScrollView,
  TouchableOpacity,
  Dimensions,
  Alert,
} from 'react-native';
import * as ImagePicker from 'expo-image-picker';
import { captureRef } from 'react-native-view-shot';
import { useRouter, usePathname, useLocalSearchParams } from 'expo-router';
import { Button } from '@/components/UI/Button';
import { TabBar } from '@/components/Navigation';
import { colors, spacing, typography, borderRadius, shadows } from '@/theme';
import { getTemplateById } from '@/data/templates';
import { useStorage } from '@/hooks/useStorage';
import { saveToGallery } from '@/utils/export';

const { width: SCREEN_WIDTH } = Dimensions.get('window');
const CANVAS_SIZE = SCREEN_WIDTH - spacing.md * 2;

export default function CreateScreen() {
  const router = useRouter();
  const pathname = usePathname();
  const params = useLocalSearchParams();
  const { saveCollage, addRecentTemplate } = useStorage();
  const canvasRef = useRef<View>(null);

  const [templateId, setTemplateId] = useState(params.templateId as string);
  const [photos, setPhotos] = useState<Record<string, string>>({});
  const [isSaving, setIsSaving] = useState(false);

  const template = templateId ? getTemplateById(templateId) : null;

  const handleSelectPhoto = async (frameId: string) => {
    try {
      const result = await ImagePicker.launchImageLibraryAsync({
        mediaTypes: ['images'],
        allowsEditing: true,
        aspect: [1, 1],
        quality: 1,
      });

      if (!result.canceled && result.assets[0]) {
        setPhotos(prev => ({ ...prev, [frameId]: result.assets[0].uri }));
      }
    } catch (error) {
      Alert.alert('Error', 'Failed to select photo');
    }
  };

  const handleRemovePhoto = (frameId: string) => {
    setPhotos(prev => {
      const newPhotos = { ...prev };
      delete newPhotos[frameId];
      return newPhotos;
    });
  };

  const handleSave = async () => {
    if (!template) {
      Alert.alert('Error', 'No template selected');
      return;
    }

    const photoCount = Object.keys(photos).length;
    if (photoCount === 0) {
      Alert.alert('Error', 'Please add at least one photo');
      return;
    }

    setIsSaving(true);
    try {
      // Capture canvas
      const uri = await captureRef(canvasRef, {
        format: 'png',
        quality: 1,
        result: 'tmpfile',
      });

      // Save to storage
      await saveCollage({
        templateId: template.id,
        thumbnail: uri,
        photos: Object.entries(photos).map(([frameId, uri]) => ({ frameId, uri })),
      });

      // Add to recent templates
      await addRecentTemplate(template.id);

      // Save to gallery
      await saveToGallery(uri);

      Alert.alert(
        'Success!',
        'Your collage has been saved to your gallery',
        [
          { text: 'OK', onPress: () => router.push('/my-creations') },
        ]
      );
    } catch (error) {
      Alert.alert('Error', 'Failed to save collage');
      console.error(error);
    } finally {
      setIsSaving(false);
    }
  };

  if (!template) {
    return (
      <View style={styles.container}>
        <View style={styles.centerContainer}>
          <Text style={styles.title}>No Template Selected</Text>
          <Button
            title="Choose a Template"
            onPress={() => router.push('/templates')}
            variant="primary"
            style={styles.button}
          />
        </View>
        <TabBar
          activeTab={pathname}
          onTabChange={(tabId) => router.push(`/${tabId === 'home' ? 'index' : tabId}`)}
        />
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <ScrollView
        style={styles.scrollView}
        contentContainerStyle={styles.scrollContent}
        showsVerticalScrollIndicator={false}
      >
        {/* Header */}
        <View style={styles.header}>
          <Text style={styles.title}>{template.name}</Text>
          <Text style={styles.subtitle}>{Object.keys(photos).length} / {template.frames.length} photos</Text>
        </View>

        {/* Canvas */}
        <View style={styles.canvasContainer}>
          <View
            ref={canvasRef}
            style={[
              styles.canvas,
              {
                aspectRatio: template.aspectRatio === '9:16' ? 9/16 : template.aspectRatio === '4:5' ? 4/5 : 1,
              },
            ]}
          >
            {template.frames.map((frame) => (
              <TouchableOpacity
                key={frame.id}
                style={[
                  styles.frame,
                  {
                    left: `${frame.x}%`,
                    top: `${frame.y}%`,
                    width: `${frame.width}%`,
                    height: `${frame.height}%`,
                  },
                ]}
                onPress={() => handleSelectPhoto(frame.id)}
              >
                {photos[frame.id] ? (
                  <>
                    {/* Photo placeholder - in production, you'd display the actual image */}
                    <View style={styles.photoPlaceholder}>
                      <Text style={styles.photoText}>‚úì</Text>
                    </View>
                    <TouchableOpacity
                      style={styles.removeButton}
                      onPress={(e) => {
                        e.stopPropagation();
                        handleRemovePhoto(frame.id);
                      }}
                    >
                      <Text style={styles.removeText}>‚úï</Text>
                    </TouchableOpacity>
                  </>
                ) : (
                  <View style={styles.emptyFrame}>
                    <Text style={styles.emptyText}>+</Text>
                  </View>
                )}
              </TouchableOpacity>
            ))}
          </View>
        </View>

        {/* Instructions */}
        <View style={styles.instructions}>
          <Text style={styles.instructionsTitle}>Instructions</Text>
          <Text style={styles.instructionsText}>
            ‚Ä¢ Tap on a frame to add a photo{'\n'}
            ‚Ä¢ Tap the X button to remove a photo{'\n'}
            ‚Ä¢ Add photos to all frames for best results
          </Text>
        </View>

        {/* Actions */}
        <View style={styles.actions}>
          <Button
            title="Save Collage"
            onPress={handleSave}
            variant="primary"
            size="large"
            fullWidth
            loading={isSaving}
            disabled={Object.keys(photos).length === 0}
          />
        </View>

        <View style={{ height: spacing.xl }} />
      </ScrollView>

      <TabBar
        activeTab={pathname}
        onTabChange={(tabId) => router.push(`/${tabId === 'home' ? 'index' : tabId}`)}
      />
    </View>
  );
}

} from 'react-native';
import { useRouter, usePathname } from 'expo-router';

import { colors, spacing, typography, borderRadius, shadows } from '@/theme';
import { useStorage } from '@/hooks/useStorage';

const { width: SCREEN_WIDTH } = Dimensions.get('window');
const CARD_WIDTH = (SCREEN_WIDTH - spacing.md * 2 - spacing.md) / 2;

export default function MyCreationsScreen() {
  const router = useRouter();
  const pathname = usePathname();
  const { collages, deleteCollage } = useStorage();
  const [selectedId, setSelectedId] = useState<string | null>(null);

  const handlePress = (collageId: string) => {
    if (selectedId === collageId) {
      // Show options
      Alert.alert(
        'Collage Options',
        'What would you like to do with this collage?',
        [
          { text: 'Cancel', style: 'cancel' },
          {
            text: 'Delete',
            style: 'destructive',
            onPress: () => handleDelete(collageId),
          },
        ],
      );
      setSelectedId(null);
    } else {
      setSelectedId(collageId);
    }
  };

  const handleDelete = async (collageId: string) => {
    Alert.alert(
      'Delete Collage',
      'Are you sure you want to delete this collage?',
      [
        { text: 'Cancel', style: 'cancel' },
        {
          text: 'Delete',
          style: 'destructive',
          onPress: async () => {
            await deleteCollage(collageId);
            setSelectedId(null);
          },
        },
      ]
    );
  };

  if (collages.length === 0) {
    return (
      <View style={styles.container}>
        <View style={styles.emptyContainer}>
          <Text style={styles.emptyTitle}>No Creations Yet</Text>
          <Text style={styles.emptyText}>
            Start creating beautiful collages and they'll appear here

            style={styles.emptyButton}

          activeTab={pathname}
          onTabChange={(tabId) => router.push(`/${tabId === 'home' ? 'index' : tabId}`)}
        />

  }

  return (
    <View style={styles.container}>
      <ScrollView
        style={styles.scrollView}
        contentContainerStyle={styles.scrollContent}
        showsVerticalScrollIndicator={false}
      >
        {/* Header */}
        <View style={styles.header}>
          <Text style={styles.title}>My Creations</Text>
          <Text style={styles.subtitle}>{collages.length} collages</Text>
        </View>

              key={collage.id}
              style={[

                selectedId === collage.id && styles.cardSelected,
              ]}
              onPress={() => handlePress(collage.id)}
              onLongPress={() => setSelectedId(collage.id)}
            >
              <View style={styles.cardPreview}>
                {/* Thumbnail placeholder */}
                <View style={styles.thumbnailPlaceholder}>
                  <Text style={styles.thumbnailText}>üñºÔ∏è</Text>
                </View>
                <View style={styles.cardBadge}>
                  <Text style={styles.cardBadgeText}>
                    {format(new Date(collage.createdAt), 'MMM d')}
                  </Text>
                </View>
              </View>
              <View style={styles.cardInfo}>
                <Text style={styles.cardName}>Collage #{collage.id.slice(-4)}</Text>
